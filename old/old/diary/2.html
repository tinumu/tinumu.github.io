#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,m;
    pair<int,int>data[10001];
    bool flag[501] = {0};
    flag[1] = true;
    vector<int>v;
    int sum = 0;
    cin >> n >> m;
    for(int i = 0;i < m;i++){
        cin >> data[i].first >> data[i].second;
        if(data[i].first == 1){
            v.push_back(data[i].second);
            flag[data[i].second] = true;
            sum++;
        }
    }
    for(int i = 0;i < v.size();i++){
        for(int j = 0;j < m;j++){
            if(data[j].first == v[i]){
                if(flag[data[j].second] == false){
                    sum++;
                    flag[data[j].second] = true;
                }
            }
            if(data[j].second == v[i]){
                if(flag[data[j].first] == false){
                    sum++;
                    flag[data[j].first] = true;
                }
            }
        }
    }
    cout << sum << endl;
    return 0;
}
